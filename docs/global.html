Operation
Example

   { data: state => {
      const patient = state.data.body;
       return {
         firstName: patient.Patient_name.split(' ')[0],
         lastName: patient.Patient_name.split(' ')[1],
         visualId: patient.Case_ID,
         'age:years': patient.Age_in_year,
